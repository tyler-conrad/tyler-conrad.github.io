Future<ui.Image> perlin4Image({
  required ui.Size size,
  required double scale,
  required double brightness,
  required vm.Vector4 xyzw,
  double elapsed = 0.0,
}) async {
  final shader = await load('lib/src/shaders/perlin4.frag');
  return toImage(
    np.Noise4Painter(
      shader: shader,
      size: size,
      elapsed: elapsed,
      xyzw: xyzw,
      scale: scale,
      brightness: brightness,
    ),
  );
}
